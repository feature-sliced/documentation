import { CHANNELS, type TElectronAPI } from 'shared/ipc';

    [CHANNELS.GET_USER_DATA]: () => ipcRenderer.sendSync(CHANNELS.GET_USER_DATA),

export const sendUser = () => {
    ipcMain.on(CHANNELS.GET_USER_DATA, ev => {

            email: 'john.doe@example.com',

    });
};
